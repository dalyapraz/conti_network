{
  this.data = WrapperX.MemAlloc(Size, 1);
  this.Offsets = [];
  this.Size = Size;
  Sizes = '1bc2nt4luf8mqd0hpwszv';
  o = 0;
  for (i = 0; i < this.Types.length; i++) // init offsets
  {
    j = Sizes.indexOf(this.Types.charAt(i));
    if (j == -1) throw new Error('In TStruct: no such type.');
    this.Offsets[i] = o; // note: field offset cannot be greater than 255
    o += (j <= 2) ? 1 :
      (j <= 5) ? 2 :
        (j <= 9) ? 4 :
          (j <= 13) ? 8 :
            PSize;
  }

  /// as js has no explicit destructor so use this func
  this.Free = function ()
  {
    WrapperX.MemFree(this.data);
  }

  this.SetField = function (Name, Value)
  {
    // WScript.Echo(Name + " = " + Value + " at addr " + this.Fields[Name][0] + ' of type ' + this.Fields[Name][1]);
    // NumPut(Var, Address[, Offset][, Type]) — writes a number to memory.

    WrapperX.NumPut(Value, this.data, this.Offsets[n], this.Types.charAt(n));//this.Types[this.Fields[Name]]);

  {
    n = this.Fields[Name];
    return WrapperX.NumGet(this.data, this.Offsets[n], this.Types.charAt(n));
  }
}
